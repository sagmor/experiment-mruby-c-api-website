179  { \
180  khint_t k, del_k, step = 0; \
181  if (h->n_occupied >= khash_upper_bound(h)) { \
182  kh_resize_##name(mrb, h, h->n_buckets*2); \
183  } \
184  k = __hash_func(mrb,key) & khash_mask(h); \
185  del_k = kh_end(h); \
186  while (!__ac_isempty(h->ed_flags, k)) { \
187  if (!__ac_isdel(h->ed_flags, k)) { \
188  if (__hash_equal(mrb,h->keys[k], key)) { \
189  if (ret) *ret = 0; \
190  return k; \
191  } \
192  } \
193  else if (del_k == kh_end(h)) { \
194  del_k = k; \
195  } \
196  k = (k+(++step)) & khash_mask(h); \
197  } \
198  if (del_k != kh_end(h)) { \
