296 /* required arguments */
297 #define MRB_ARGS_REQ(n) ((mrb_aspec)((n)&0x1f) << 18)
298 /* optional arguments */
299 #define MRB_ARGS_OPT(n) ((mrb_aspec)((n)&0x1f) << 13)
300 /* mandatory and optinal arguments */
301 #define MRB_ARGS_ARG(n1,n2) (MRB_ARGS_REQ(n1)|MRB_ARGS_OPT(n2))

303 /* rest argument */
304 #define MRB_ARGS_REST() ((mrb_aspec)(1 << 12))
305 /* required arguments after rest */
306 #define MRB_ARGS_POST(n) ((mrb_aspec)((n)&0x1f) << 7)
307 /* keyword arguments (n of keys, kdict) */
308 #define MRB_ARGS_KEY(n1,n2) ((mrb_aspec)((((n1)&0x1f) << 2) | ((n2)?(1<<1):0)))
309 /* block argument */
310 #define MRB_ARGS_BLOCK() ((mrb_aspec)1)

312 /* accept any number of arguments */
313 #define MRB_ARGS_ANY() MRB_ARGS_REST()

486 #ifndef ISPRINT
487 #define ISASCII(c) ((unsigned)(c) <= 0x7f)
488 #define ISPRINT(c) (((unsigned)(c) - 0x20) < 0x5f)
489 #define ISSPACE(c) ((c) == ' ' || (unsigned)(c) - '\t' < 5)
490 #define ISUPPER(c) (((unsigned)(c) - 'A') < 26)
491 #define ISLOWER(c) (((unsigned)(c) - 'a') < 26)
492 #define ISALPHA(c) ((((unsigned)(c) | 0x20) - 'a') < 26)
493 #define ISDIGIT(c) (((unsigned)(c) - '0') < 10)
494 #define ISXDIGIT(c) (ISDIGIT(c) || ((unsigned)(c) | 0x20) - 'a' < 6)
495 #define ISALNUM(c) (ISALPHA(c) || ISDIGIT(c))
496 #define ISBLANK(c) ((c) == ' ' || (c) == '\t')
497 #define ISCNTRL(c) ((unsigned)(c) < 0x20 || (c) == 0x7f)
498 #define TOUPPER(c) (ISLOWER(c) ? ((c) & 0x5f) : (c))
